How to Bulk Check Domain Availability Using Node.js

Having to bulk check domain availability has become a bit of a necessity when setting up a new website, especially for .com domains.

Most of the websites that allow you to check domain availablility only let you check one domain at a time. They can also be quite slow and push adverts for .neverheardof TLDs.

I also wanted to easily test for differing word orders in potential domain names. If I wanted a domain about pink elephants, "pinkelephants.com", I also wanted to test for a domain with the two words reversed (£elephantspink.com") incase my first choice was unavailable. Of course, a lot of the time the words reversed would be plain dumb, but there is always potential for an unforseen fortuitous discovery.

My second requirement was to save the results to a text file for importing into Excel so I can have a record of what I searched and whether they were available.

The first step is being able to query the whois database. It could be possible to call one of the many websites that allow you to do domain searches and scrape the data. But this could potentially cause problems with bulk queries and triggering the sites denial of service defences!

But there is a relatively simple solution using Node.js. Node.js has a NPM package available called whois (formally "node-whois"). This does a whois lookup for any domain name using the whois protocol, which operates over TCP port 43. This makes querying a whois database straight forward. Simply require the package and call the method "lookup" with the domain name your are looking for and a callback to handle the returned response.


    var whois = require('whois');
    whois.lookup("pinkelephants.com", function (err, data) {
            if (err) {
                console.error(err);
            } else {
                //do something with data
                //which will just be a typical whois text output
            }

    });

Typically the data returned for a .com query where a domain is already registered will be something like this:

Domain Name: PINKELEPHANTS.COM Registry Domain ID: 104511878_DOMAIN_COM-VRSN Registrar WHOIS Server: whois.uniregistrar.com Registrar URL: http://www.uniregistrar.com Updated Date: 2017-11-29T15:22:08Z Creation Date: 2003-10-03T18:26:29Z Registry Expiry Date: 2019-10-03T18:26:29Z ...

Where the domain is NOT registered it will be:

No match for domain "ELEPHANTSPINK.COM". ...

In the returned result I search for the phrase on the first line of: "No match for domain".

Read List of Potential Domain Names from a Text File

The next step is to put a list of my potential domain names in a text file which my script will load and loop through and call whois.lookup. However, since I want to do the word swapping thing I'll stick a | (pipe) within a name to tell the script where to split the two words and perform a swap.

pink|elephants
badger|cat
dontswap
threewordshere

And the full code:


    var whois = require('whois');
    var fs = require('fs');
    vr path = require('path');

    function doLookup(domain, cb) {
        whois.lookup(domain, function (err, data) {
            console.log("lookup returned");
            if (err) {
                console.error(err);
            } else {
                cb(err, data, domain)
            }
        });
    }
    var noMatch = new RegExp(/No match for domain/);

    var txt = fs.readFileSync("./domains.txt", "utf-8");
    var items = txt.split("\n");
    var domainList = [];
    function addDomain(name) {
        if (name) {
            domainList.push(name + ".com");
            console.log(name + " added");
        } else {
            console.log("cannot add blank name...");
        }
    }
    var i = 0;
    for (i = 0; i < items.length; i++) {
        var words = items[i].split("|");
        if (words.length > 1) {
            addDomain(words[0] + words[1]);
            addDomain(words[1] + words[0]);
        } else {
            addDomain(words[0])
        }
    }

    var k = 0;
    var output = [];
    var writeResults = true;
    function onReturn(err, data, domain) {
        var lines = data.split('\n');
        var firstLine = lines[0];
        console.log(firstLine);
        var m = firstLine.match(noMatch);
        var available = m ? true : false;
        console.log(domain + ": " + available);
        output.push(domain + ", " + available);
        if (writeResults) {
            fs.writeFileSync(domain + "-result.txt", data, 'utf-8');
        }
        k++;
        if (k < domainList.length) {
            doLookup(domainList[k], onReturn);
        } else {
            var txtfile = domainList[0] + "-" + domainList[domainList.length - 1] + ".txt";
            var txt = output.join("\n");
            fs.writeFileSync(txtfile, txt, "utf-8");
            console.log(domainList.length + " domains processed");
        }
    }

    doLookup(domainList[0], onReturn);
